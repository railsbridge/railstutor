qvalues http header parser code taken from webrick

[Source]

     # File app/controllers/application.rb, line 230
230:   def parse_qvalues(value)
231:     tmp = []
232:     if value
233:       parts = value.split(/,\s*/)
234:       parts.each {|part|
235:         if m = %r{^([^\s,]+?)(?:;\s*q=(\d+(?:\.\d+)?))?$}.match(part)
236:           val = m[1]
237:           q = (m[2] or 1).to_f
238:           tmp.push([val, q])
239:         end
240:       }
241:       tmp = tmp.sort_by{|val, q| -q}
242:       tmp.collect!{|val, q| val}
243:     end
244:     return tmp
245:   rescue
246:     nil
247:   end
